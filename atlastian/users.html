const userListDiv = document.getElementById('user-list'); // Example API (or you can use your own) const API_URL = 'https://jsonplaceholder.typicode.com/users'; // Function to fetch data async function fetchData() { try { const response = await fetch(API_URL); let users = await response.json(); // Check if there's saved data in localStorage const savedUsers = localStorage.getItem('users'); if (savedUsers) { users = JSON.parse(savedUsers); } renderUsers(users); } catch (error) { console.error('Error fetching data:', error); } } // Function to render users in the DOM function renderUsers(users) { userListDiv.innerHTML = ''; // Clear previous content users.forEach((user, index) => { const div = document.createElement('div'); div.className = 'item'; // Input for editing const input = document.createElement('input'); input.type = 'text'; input.value = user.name; // Save button const saveBtn = document.createElement('button'); saveBtn.textContent = 'Save'; saveBtn.addEventListener('click', () => { users[index].name = input.value; // Update local data localStorage.setItem('users', JSON.stringify(users)); // Save locally alert('Saved locally!'); }); div.appendChild(input); div.appendChild(saveBtn); userListDiv.appendChild(div); }); } // Initial fetch fetchData();
